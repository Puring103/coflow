# coflow语法总览

本文档用于快速展示coflow当前语法草案

# 模块

```coflow
import common
import weapons
import weapons as w
```

一个文件是一个模块
顶层声明默认公开
使用local声明文件内私有成员

```coflow
local var internal_scale = 2

local fn helper() -> int {
  return 10
}
```

# 类型与配置

```coflow
enum Rarity {
  common
  rare
  epic
}

class Weapon {
  id: string
  name: string
  rarity: Rarity
  damage: int
  cooldown: float = 1.0

  fn validate(self) -> null | string | [string] {
    if self.damage <= 0 {
      return "damage must be greater than 0"
    }

    return null
  }
}
```

顶层`name = value`是配置
顶层`name: Type = value`是带类型标注的配置
配置必须在编译期确定

```coflow
base_sword: Weapon = {
  id: "base_sword",
  name: "Base Sword",
  rarity: Rarity.common,
  damage: 10,
}

strong_sword: Weapon = {
  ...base_sword,
  id: "strong_sword",
  name: "Strong " + base_sword.name,
  damage: base_sword.damage + 5,
}
```

# 函数

```coflow
fn add(a: int, b: int) -> int {
  return a + b
}

fn spawn(id: string, x: float = 0, y: float = 0) {
  host.spawn(id, x, y)
}

fn print_value(value) {
  print(value)
}

var add_one = fn(x: int) -> int {
  return x + 1
}
```

支持命名参数

```coflow
spawn(id: "slime", x: 10, y: 5)
spawn("slime", y: 10)
```

# 协程

```coflow
co fn wait_click() {
  yield wait.event("click")
}

co fn dialog() {
  dialog.show("hello")
  yield wait_click()
  dialog.show("world")
}

var c = dialog()
c.resume()
```

co fn创建协程上下文
yield只能直接出现在co fn函数体内
普通fn不能yield
co fn必须至少包含一个yield
yield另一个协程时自动委托执行

```coflow
co fn child() {
  yield 1
  yield 2
}

co fn parent() {
  yield child()
  yield 3
}
```
